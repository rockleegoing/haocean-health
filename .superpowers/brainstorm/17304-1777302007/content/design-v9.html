// ChapterTreeItem 新增 Article 层级
sealed class ChapterTreeItem {
    data class Chapter(
        val chapterId: Long,
        val chapterNo: String?,
        val chapterTitle: String?,
        var isExpanded: Boolean = false,
        var legalBasisCount: Int = 0,
        var processingBasisCount: Int = 0,
        var articles: List<Article> = emptyList()
    ) : ChapterTreeItem()

    data class Article(
        val articleId: Long,
        val chapterId: Long?,
        val articleNo: String?,
        val content: String?,
        var isExpanded: Boolean = false,
        var legalBasisCount: Int = 0,      // 新增
        var processingBasisCount: Int = 0    // 新增
    ) : ChapterTreeItem()
}

交互说明

  • 点击章节行：展开/收起该章节下的所有条目
  • 点击条目行：展开/收起条目详情内容
  • 点击条目上的"定性依据 N"：跳转到定性依据列表（只显示该条目关联的）
  • 点击条目上的"处理依据 N"：跳转到处理依据列表（只显示该条目关联的）
  • 点击章节上的"定性依据 N"：跳转到定性依据列表（显示该章节下所有条目关联的汇总）

请确认这个结构是否符合预期。
